```c
/* ---- writer (JsonW): a growable emit buffer with per-depth comma
   discipline. Counterpart to the parser above; json_test.c proves the two
   agree. Aborts on OOM — callers are the server and the sync journal, both
   of which prefer death to silent truncation. */

static void jw_grow(JsonW *w, size_t need) {
    char *nb;
    if (w->len + need + 1 <= w->cap) return;
    while (w->cap < w->len + need + 1) w->cap *= 2;
    nb = (char *)realloc(w->buf, w->cap);
    if (!nb) abort();
    w->buf = nb;
}

static void jw_puts(JsonW *w, const char *s) {
    size_t n = strlen(s);
    jw_grow(w, n);
    memcpy(w->buf + w->len, s, n);
    w->len += n;
    w->buf[w->len] = 0;
}

/* comma before a value/key if the current container already has one */
static void jw_comma(JsonW *w) {
    if (w->depth > 0 && w->count[w->depth - 1] > 0) jw_puts(w, ",");
    if (w->depth > 0) w->count[w->depth - 1]++;
}

void jw_init(JsonW *w) {
    w->cap = 256; w->len = 0; w->depth = 0;
    w->buf = (char *)malloc(w->cap);
    if (!w->buf) abort();
    w->buf[0] = 0;
}

void jw_free(JsonW *w) { free(w->buf); w->buf = NULL; w->len = w->cap = 0; }

void jw_obj_open(JsonW *w) {
    jw_comma(w);
    jw_puts(w, "{");
    if (w->depth < JW_DEPTH_MAX) w->count[w->depth] = 0;
    w->depth++;
}
void jw_obj_close(JsonW *w) { w->depth--; jw_puts(w, "}"); }
void jw_arr_open(JsonW *w) {
    jw_comma(w);
    jw_puts(w, "[");
    if (w->depth < JW_DEPTH_MAX) w->count[w->depth] = 0;
    w->depth++;
}
void jw_arr_close(JsonW *w) { w->depth--; jw_puts(w, "]"); }

static void jw_escaped(JsonW *w, const char *s) {
    const char *p;
    char buf[8];
    jw_puts(w, "\"");
    for (p = s; *p; p++) {
        unsigned char c = (unsigned char)*p;
        switch (c) {
        case '"':  jw_puts(w, "\\\""); break;
        case '\\': jw_puts(w, "\\\\"); break;
        case '\n': jw_puts(w, "\\n");  break;
        case '\r': jw_puts(w, "\\r");  break;
        case '\t': jw_puts(w, "\\t");  break;
        default:
            if (c < 0x20) { sprintf(buf, "\\u%04x", (unsigned)c); jw_puts(w, buf); }
            else { buf[0] = (char)c; buf[1] = 0; jw_puts(w, buf); }  /* UTF-8 passes through */
        }
    }
    jw_puts(w, "\"");
}

void jw_key(JsonW *w, const char *key) {
    jw_comma(w);
    jw_escaped(w, key);
    jw_puts(w, ":");
    /* the value that follows must not re-comma: */
    if (w->depth > 0) w->count[w->depth - 1]--;
}

void jw_str(JsonW *w, const char *s) { jw_comma(w); jw_escaped(w, s); }

void jw_num_long(JsonW *w, long v) {
    char buf[32];
    jw_comma(w);
    sprintf(buf, "%ld", v);
    jw_puts(w, buf);
}

void jw_num_double(JsonW *w, double v) {
    char buf[40];
    jw_comma(w);
    sprintf(buf, "%.10g", v);
    jw_puts(w, buf);
}

void jw_bool(JsonW *w, int b) { jw_comma(w); jw_puts(w, b ? "true" : "false"); }
void jw_null(JsonW *w)        { jw_comma(w); jw_puts(w, "null"); }
void jw_raw(JsonW *w, const char *json) { jw_comma(w); jw_puts(w, json); }
const char *jw_cstr(const JsonW *w) { return w->buf; }
```

static int read_body(struct mg_connection *c, char *buf, int cap) {

    buf[n < cap ? n : cap - 1] = 0;   /* NUL for json_parse; body is text */

/* ---- op validation (exported for srv_sync_test) -------------------------- */

int sync_op_valid(const JsonValue *op) {
    const char *client_op, *kind, *nid, *o;
    const JsonValue *payload;
    if (!op || op->type != JSON_OBJECT) return 0;
    client_op = json_string(json_member(op, "client_op"));
    kind      = json_string(json_member(op, "entity_kind"));
    nid       = json_string(json_member(op, "entity_nid"));
    o         = json_string(json_member(op, "op"));
    payload   = json_member(op, "payload");
    if (!client_op || !nid_ok(client_op)) return 0;
    if (!nid || !nid_ok(nid)) return 0;
    if (!kind || (strcmp(kind, "board") != 0 && strcmp(kind, "card") != 0)) return 0;
    if (!o || (strcmp(o, "create") != 0 && strcmp(o, "update") != 0
            && strcmp(o, "delete") != 0)) return 0;
    if (!payload || payload->type != JSON_OBJECT) return 0;
    /* payload size cap: re-serialize is wasteful; bound the TEXT field (the
       only unbounded one) and member count instead */
    if (json_count(payload) > 64) return 0;
    {
        const char *text = json_string(json_member(payload, "text"));
        if (text && strlen(text) > SYNC_PAYLOAD_CAP) return 0;
        text = json_string(json_member(payload, "title"));
        if (text && strlen(text) > 1024) return 0;
    }
    return 1;
}

/* ---- POST /sync/push ----------------------------------------------------- */

int h_sync_push(struct mg_connection *c, void *app) {
    SrvDb *db = srv_app_db(app);
    char machine[64];
    char *body;
    JsonValue *v = NULL;
    const JsonValue *ops;
    const char *world_id;
    long uid, last_id = 0;
    sol_u32 i, nops;
    int n;

    uid = bearer_device(c, db, machine);
    if (uid == 0) { send_json_error(c, 401, "Unauthorized", "device token required"); return 401; }

    body = (char *)malloc(SYNC_BODY_CAP);
    if (!body) { send_json_error(c, 500, "Internal Server Error", "oom"); return 500; }
    n = read_body(c, body, SYNC_BODY_CAP);
    if (n < 0) { free(body); send_json_error(c, 413, "Payload Too Large", "body too large"); return 413; }

    v = json_parse(body);
    free(body);
    world_id = json_string(json_member(v, "world_id"));
    ops      = json_member(v, "ops");
    if (!world_id || !nid_ok(world_id) || !ops || ops->type != JSON_ARRAY
        || json_count(ops) > SYNC_OPS_MAX) {
        json_free(v);
        send_json_error(c, 400, "Bad Request", "world_id + ops[<=100] required");
        return 400;
    }
    if (!world_known(db, world_id)) {
        json_free(v);
        send_json_error(c, 400, "Bad Request", "unknown world");
        return 400;
    }

    nops = json_count(ops);
    /* validate ALL before appending ANY (all-or-nothing per request) */
    for (i = 0; i < nops; i++) {
        if (!sync_op_valid(json_index((JsonValue *)ops, i))) {
            json_free(v);
            send_json_error(c, 400, "Bad Request", "invalid op in batch");
            return 400;
        }
    }

    for (i = 0; i < nops; i++) {
        const JsonValue *op = json_index((JsonValue *)ops, i);
        SrvEventIn ev;
        JsonW pw;
        long id = 0;
        memset(&ev, 0, sizeof ev);
        ev.actor_id      = uid;
        ev.origin_device = machine;
        ev.world_id      = world_id;
        ev.client_op     = json_string(json_member(op, "client_op"));
        ev.entity_kind   = json_string(json_member(op, "entity_kind"));
        ev.entity_nid    = json_string(json_member(op, "entity_nid"));
        ev.op            = json_string(json_member(op, "op"));
        /* re-serialize the payload subtree via the writer so the stored text
           is exactly one canonical JSON object */
        jw_init(&pw);
        json_write_value(&pw, json_member(op, "payload"));   /* Task 5 adds this tiny
            recursive DOM->JsonW emitter to json.c/h; see Step 5.3 */
        ev.payload = jw_cstr(&pw);
        if (srv_events_append(db, &ev, &id) != 0) {
            jw_free(&pw);
            json_free(v);
            send_json_error(c, 500, "Internal Server Error", "append failed");
            return 500;
        }
        jw_free(&pw);
        if (id > last_id) last_id = id;
    }
    json_free(v);

    {
        JsonW w;
        jw_init(&w);
        jw_obj_open(&w);
        jw_key(&w, "cursor"); jw_num_long(&w, last_id);
        jw_obj_close(&w);
        send_json(c, 200, "OK", &w);
        jw_free(&w);
    }
    return 200;
}

int h_sync_pull(struct mg_connection *c, void *app) {
    SrvDb *db = srv_app_db(app);
    const struct mg_request_info *ri = mg_get_request_info(c);
    char machine[64], world[NID_LEN + 8], after_s[24];
    SrvEventOut *evs = NULL;
    long uid, after = 0, cursor;
    int count = 0, i;

    uid = bearer_device(c, db, machine);
    if (uid == 0) { send_json_error(c, 401, "Unauthorized", "device token required"); return 401; }

    world[0] = 0; after_s[0] = 0;
    if (ri->query_string) {
        if (mg_get_var(ri->query_string, strlen(ri->query_string), "world",
                       world, sizeof world) < 0) world[0] = 0;
        if (mg_get_var(ri->query_string, strlen(ri->query_string), "after",
                       after_s, sizeof after_s) >= 0) after = atol(after_s);
    }
    if (!nid_ok(world)) {
        send_json_error(c, 400, "Bad Request", "world (nid) required");
        return 400;
    }

    if (srv_events_after(db, world, after, SYNC_PULL_MAX, &evs, &count) != 0) {
        send_json_error(c, 500, "Internal Server Error", "query failed");
        return 500;
    }
    cursor = count > 0 ? evs[count - 1].id : after;

    {
        JsonW w;
        jw_init(&w);
        jw_obj_open(&w);
        jw_key(&w, "cursor"); jw_num_long(&w, cursor);
        jw_key(&w, "events"); jw_arr_open(&w);
        for (i = 0; i < count; i++) {
            jw_obj_open(&w);
            jw_key(&w, "id");          jw_num_long(&w, evs[i].id);
            jw_key(&w, "ts");          jw_num_long(&w, evs[i].ts);
            jw_key(&w, "actor");       jw_num_long(&w, evs[i].actor_id);
            jw_key(&w, "origin");      jw_str(&w, evs[i].origin_device);
            jw_key(&w, "entity_kind"); jw_str(&w, evs[i].entity_kind);
            jw_key(&w, "entity_nid");  jw_str(&w, evs[i].entity_nid);
            jw_key(&w, "op");          jw_str(&w, evs[i].op);
            jw_key(&w, "payload");     jw_raw(&w, evs[i].payload);
            jw_obj_close(&w);
        }
        jw_arr_close(&w);
        jw_obj_close(&w);
        send_json(c, 200, "OK", &w);
        jw_free(&w);
    }
    srv_events_free(evs, count);
    return 200;
}

/* json.c */
void json_write_value(JsonW *w, const JsonValue *v) {
    sol_u32 i;
    if (!v) { jw_null(w); return; }
    switch (v->type) {
    case JSON_NULL:   jw_null(w); break;
    case JSON_BOOL:   jw_bool(w, v->u.boolean); break;
    case JSON_NUMBER: jw_num_double(w, v->u.number); break;
    case JSON_STRING: jw_str(w, v->u.string); break;
    case JSON_ARRAY:
        jw_arr_open(w);
        for (i = 0; i < v->u.array.count; i++) json_write_value(w, v->u.array.items[i]);
        jw_arr_close(w);
        break;
    case JSON_OBJECT:
        jw_obj_open(w);
        for (i = 0; i < v->u.object.count; i++) {
            jw_key(w, v->u.object.members[i].key);
            json_write_value(w, v->u.object.members[i].value);
        }
        jw_obj_close(w);
        break;
    }
}

static NSURLSession *g_session = nil;

void http_init(void) {
    if (g_session) return;
    NSURLSessionConfiguration *cfg = [NSURLSessionConfiguration ephemeralSessionConfiguration];
    cfg.timeoutIntervalForRequest = 15.0;
    g_session = [NSURLSession sessionWithConfiguration:cfg];
}

static void push_result(int rid, int status, NSData *data) {
    pthread_mutex_lock(&g_lock);
    if (g_count < HTTP_QUEUE_CAP) {
        HttpResult *r = &g_queue[g_count++];
        r->id = rid;
        r->status = status;
        r->body_len = data ? (long)data.length : 0;
        r->body = (char *)malloc((size_t)r->body_len + 1);
        if (r->body) {
            if (data && r->body_len > 0) memcpy(r->body, data.bytes, (size_t)r->body_len);
            r->body[r->body_len] = 0;
        } else {
            r->body_len = 0;
        }
    }
    pthread_mutex_unlock(&g_lock);
}

void http_request(int rid, const char *url, const char *bearer,
                  const char *json_body) {
    NSURL *u;
    NSMutableURLRequest *req;
    if (!g_session) http_init();
    u = [NSURL URLWithString:[NSString stringWithUTF8String:url]];
    if (!u) { push_result(rid, 0, nil); return; }
    req = [NSMutableURLRequest requestWithURL:u];
    if (bearer) {
        [req setValue:[NSString stringWithFormat:@"Bearer %s", bearer]
             forHTTPHeaderField:@"Authorization"];
    }
    if (json_body) {
        req.HTTPMethod = @"POST";
        [req setValue:@"application/json" forHTTPHeaderField:@"Content-Type"];
        req.HTTPBody = [NSData dataWithBytes:json_body length:strlen(json_body)];
    }
    [[g_session dataTaskWithRequest:req
                  completionHandler:^(NSData *data, NSURLResponse *resp, NSError *err) {
        int status = 0;
        if (!err && [resp isKindOfClass:[NSHTTPURLResponse class]])
            status = (int)((NSHTTPURLResponse *)resp).statusCode;
        push_result(rid, status, err ? nil : data);
    }] resume];
}

int http_poll(HttpResult *out, int max) {
    int n;
    pthread_mutex_lock(&g_lock);
    n = g_count < max ? g_count : max;
    if (n > 0) {
        memcpy(out, g_queue, (size_t)n * sizeof(HttpResult));
        memmove(g_queue, g_queue + n, (size_t)(g_count - n) * sizeof(HttpResult));
        g_count -= n;
    }
    pthread_mutex_unlock(&g_lock);
    return n;
}

   transport (production: platform_http; tests: a fake). Owns push batching,
   pull cadence, backoff, cursor/journal state via syncstore. Never touches
   the Scene — pulled event batches go to the app through the apply callback
   (spec §7/§8). */
#ifndef SYNC_H
#define SYNC_H
#include "sol_base.h"
#include "syncstore.h"

typedef struct {
    int   id;
    int   status;
    char *body;
    long  body_len;
} SyncHttpResult;   /* layout-identical to platform_http.h's HttpResult */

typedef struct {
    void (*request)(int id, const char *url, const char *bearer, const char *json_body);
    int  (*poll)(SyncHttpResult *out, int max);
} SyncTransport;

typedef enum {
    SYNC_OFF = 0, SYNC_IDLE, SYNC_PUSHING, SYNC_PULLING, SYNC_BACKOFF, SYNC_SIGNED_OUT
} SyncState;

#define SYNC_PUSH_BATCH   100
#define SYNC_PULL_PERIOD  30.0
#define SYNC_BACKOFF_MAX  300.0
#define SYNC_ERR_CAP      160

/* the app applies a pull batch (the events-array JSON verbatim); returns
   SOL_TRUE if applied (cursor may then advance) */
typedef sol_bool (*SyncApplyFn)(void *ctx, const char *events_json);

typedef struct {
    SyncTransport t;
    SyncStore    *store;
    char          world_id[NID_LEN + 1];
    char          url[SYNCSTORE_URL_CAP];     /* base, no trailing slash */
    char          token[64];
    SyncState     state;
    long          cursor;
    double        next_pull_at, retry_at, backoff;
    int           req_seq, inflight_id, inflight_lines;
    char          last_error[SYNC_ERR_CAP];
    double        last_push_t, last_pull_t;
    sol_bool      wake;                        /* journal grew / manual */
} SyncEngine;

void sync_init(SyncEngine *e, SyncTransport t, SyncStore *store,
               const char *world_id, const char *url, const char *token);
/* one frame: poll transport, advance the machine. now = seconds (monotonic). */
void sync_tick(SyncEngine *e, double now, SyncApplyFn apply, void *ctx);
void sync_wake(SyncEngine *e);        /* journal grew or user hit Sync now */
/* one status line for the HUD/palette, e.g. "idle · 3 queued · pulled 12s ago" */
void sync_status(const SyncEngine *e, double now, char *out, int cap);

```

  Engine behavior contract (what the tests pin):
  1. init loads cursor from the store; state IDLE (caller only constructs it when sync is on + signed in).
  2. Tick with a non-empty journal (or wake) → POST `{base}/sync/push` with up to SYNC_PUSH_BATCH journal lines spliced via `jw_raw`; state PUSHING; remembers how many lines are in flight.
  3. Push 200 → `syncstore_journal_truncate(inflight_lines)`, then immediately begin a pull. Push 401 → SIGNED_OUT (journal untouched). Push 4xx≠401 → BACKOFF with the error text kept (a poisoned journal should be visible, not spin). Push 0/5xx → BACKOFF (exponential ×2 from 5s, capped 300; journal untouched).
  4. Pull: GET `{base}/sync/pull?world=W&after=<cursor>`. 200 → hand the `events` array text to `apply`; on SOL_TRUE, `syncstore_cursor_save(new cursor)` and cursor updates in memory; if the batch was full (== 500 events guessed by `cursor jump`? NO — the response has no "more" flag; detect a full batch by counting events via json_count and pull again immediately when count == 500). 401 → SIGNED_OUT. Else → BACKOFF.
  5. Idle pulls every SYNC_PULL_PERIOD; any success resets backoff to 5s base.
  6. Exactly one request in flight ever (inflight_id nonzero → only poll).

/* Apply one pulled event to the scene. Known-board events mutate directly;
   events for boards this world lacks (and their cards) PARK in the arrivals
   file (spec §9). Runs with sync_apply_active set — app_scene_save journals
   nothing. Returns SOL_TRUE if handled (parked counts as handled). */
static sol_bool sync_apply_one(AppState *st, const JsonValue *ev);

static sol_bool sync_apply_events_cb(void *ctx, const char *events_json) {
    AppState *st = (AppState *)ctx;
    JsonValue *arr = json_parse(events_json);
    sol_u32 i, n;
    sol_bool all_ok = SOL_TRUE;
    if (!arr || arr->type != JSON_ARRAY) { json_free(arr); return SOL_FALSE; }
    st->sync_apply_active = SOL_TRUE;
    n = json_count(arr);
    for (i = 0; i < n; i++) {
        const JsonValue *ev = json_index(arr, i);
        const char *origin = json_string(json_member(ev, "origin"));
        if (origin && strcmp(origin, st->sync_machine) == 0) continue;  /* echo */
        if (!sync_apply_one(st, ev)) all_ok = SOL_FALSE;
    }
    if (n > 0) app_scene_save(st);   /* one save per batch; hook suppressed */
    st->sync_apply_active = SOL_FALSE;
    if (all_ok) {   /* the shadow must absorb remote changes so the NEXT local
                       save doesn't re-diff them into the journal */
        snap_free(&st->sync_shadow);
        snap_take(&st->scene, &st->sync_shadow);
    }
    json_free(arr);
    return all_ok;
}
```
